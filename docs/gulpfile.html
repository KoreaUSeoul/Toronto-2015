    css     : src + '/styles/*.css'
};

    index : src,
    css   : src + '/styles',
    docs  : './docs'

    gulp.task('inject:css', function() {
    var sources = gulp.src(globs.css, {read: false});    

    return gulp
    .src(globs.index)
    .pipe(inject(sources, {relative: true}))
    .pipe(gulp.dest(dests.index));
});
