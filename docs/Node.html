    class Node
    constructor: (@x, @y, @r, @ctx) ->
        @hover = false

    draw: ->
        @ctx.beginPath()
        @ctx.moveTo(@x, @y)
        @ctx.arc(@x, @y, @r, 0, 2 * Math.PI)
        @ctx.closePath()

        if @hover
            @ctx.fillStyle = "red"
        else
            @ctx.fillStyle = "black"
